# 实现文本按照一定高度展开与收起

# 需求要点

  • 文本高度固定
  • 文本内容长度不固定，可能不能撑满，也可能溢出。
  • 若不能撑满，以最小高度展示
  • 若溢出，展示「展开」按钮，点击展开全部文本，并展示「收起」按钮，点击「收起」按钮，恢复原样
# 实现方式

demo地址：https://github.com/dandanDQ/wonderful-demos

<template>
  <div class="expand-collapse-panel">
    <input v-if="showMoreBtn" type="checkbox" :id="id"/>
    <div class="text-area" :ref="id" :style="{'--height':height+'px'}">{{text}}</div>
     <label v-if="showMoreBtn" :for="id" :data-expandtext="expandText" :data-collaspetext="collapseText"></label>
  </div>
</template>
<script>
export default {
  name: 'expand-collapse-panel',
  props: {
    height: {
      type: Number,
      default: 100
    },
    text: {
      type: String,
      default: ''
    },
    collapseText: {
      type: String,
      default: 'collapse'
    },
    expandText: {
      type: String,
      default: 'expand'
    },
    id: {
      type: String,
      default: 'defaultref'
    }
  },
  data(){
    return {
      showMoreBtn: false,
    }
  },
  mounted(){
    this.checkOverflow()
  },
  watch:{
    text:{
      handler:function (){
        this.checkOverflow()
      },
    }
  },
  methods: {
    checkOverflow () {
      let el = this.$refs[this.id]
      if (el) {
        let isOverflowing = el.scrollHeight > this.height
        if (!isOverflowing && el.style) {
          el.style.height = 'auto'
        }
        this.showMoreBtn = isOverflowing
        return
      }
      this.showMoreBtn = false
      return
    }
  }
}
</script>
